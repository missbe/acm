题目描述
输入三个字符后，按各字符的ASCII码从小到大的顺序输出这三个字符。

输入
输入数据有多组，每组占一行，有三个字符组成，之间无空格。

输出
对于每组输入数据，输出一行，字符中间用一个空格分开。

样例输入
qwe
asd
zxc
样例输出
e q w
a d s
c x z
//CODE
#include<cstdio>
#include<cmath>

int main(){
	const int N=3;
	char ch[N],temp;
	int i,j,k,test;
	while(scanf("%d",&test)!=EOF){
	//	ch[0]=a,ch[1]=b,ch[2]=c;
		scanf("%c%c%c",&ch[0],&ch[1],&ch[2]);
		for(i=0;i<N-1;i++){
		    k=i;
			for(j=i+1;j<N;j++)
				if(ch[j]<ch[k])
					k=j;
			if(i!=k){
				temp=ch[i];
				ch[i]=ch[k];
				ch[k]=temp;
			}
		}
		for(i=0;i<N;i++){
			if(i==N-1)
				printf("%c\n",ch[i]);
			else
				printf("%c ",ch[i]);
		}
		fflush(stdin);
	}
     return 0;
}
